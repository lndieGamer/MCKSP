# (см. common/decisions/SOV_trust_decisions.txt). Дальше: .2 приказ/отмена →
# (45 дней) → .4 расстрел (50%) / .5 побег за границу (50%).
# Флаги: mcksp_scapegoat_in_progress (выбор), mcksp_scapegoat_committed

# Подготовка к задержанию: отдать приказ или отменить операцию.
country_event = {
	id = scapegoat_event.2
	title = scapegoat_event.2.t
	desc = scapegoat_event.2.d
	picture = GFX_report_event_soviet_purge_officers_02

	is_triggered_only = yes

	# Отдать приказ — НЕОБРАТИМО. Через 45 дней 50/50: расстрел или побег.
	# Имя кнопки белое; красное предупреждение — в тултипе. Исходы скрыты
	# (hidden_effect), чтобы шансы и id ивентов не светились игроку.
	option = {
		name = scapegoat_event.2.order
		custom_effect_tooltip = scapegoat_event.2.order_tt
		hidden_effect = {
			random_list = {
				50 = { country_event = { id = scapegoat_event.4 days = 45 } }
				50 = { country_event = { id = scapegoat_event.5 days = 45 } }
			}
		}
	}

	# Отменить операцию — возврат на исходные, решение снова доступно.
	option = {
		name = scapegoat_event.2.cancel
		MCKSP_scapegoat_clear_flags = yes
	}
}
