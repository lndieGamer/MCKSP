# ----------------------------------------------------------------------------
# Геометрия. ax/ay = точка A, bx/by = точка B (temp). Результат в mcksp_tmp_*.
# ----------------------------------------------------------------------------
MCKSP_calc_dist2 = {
	set_temp_variable = { mcksp_tmp_dx = mcksp_tmp_bx }
	subtract_from_temp_variable = { mcksp_tmp_dx = mcksp_tmp_ax }
	set_temp_variable = { mcksp_tmp_dy = mcksp_tmp_by }
	subtract_from_temp_variable = { mcksp_tmp_dy = mcksp_tmp_ay }
	multiply_temp_variable = { mcksp_tmp_dx = mcksp_tmp_dx }
	multiply_temp_variable = { mcksp_tmp_dy = mcksp_tmp_dy }
	set_temp_variable = { mcksp_tmp_dist2 = mcksp_tmp_dx }
	add_to_temp_variable = { mcksp_tmp_dist2 = mcksp_tmp_dy }
}

MCKSP_calc_manhattan = {
	set_temp_variable = { mcksp_tmp_dx = mcksp_tmp_bx }
	subtract_from_temp_variable = { mcksp_tmp_dx = mcksp_tmp_ax }
	set_temp_variable = { mcksp_tmp_dy = mcksp_tmp_by }
	subtract_from_temp_variable = { mcksp_tmp_dy = mcksp_tmp_ay }
	if = { limit = { check_variable = { mcksp_tmp_dx < 0 } } multiply_temp_variable = { mcksp_tmp_dx = -1 } }
	if = { limit = { check_variable = { mcksp_tmp_dy < 0 } } multiply_temp_variable = { mcksp_tmp_dy = -1 } }
	set_temp_variable = { mcksp_tmp_manh = mcksp_tmp_dx }
	add_to_temp_variable = { mcksp_tmp_manh = mcksp_tmp_dy }
}

# Координаты текущего "я" игрока -> ax/ay.
MCKSP_load_coords_of_player = {
	if = { limit = { check_variable = { player_self = 1 } }  set_temp_variable = { mcksp_tmp_ax = -2 }   set_temp_variable = { mcksp_tmp_ay = -5 } }
	else_if = { limit = { check_variable = { player_self = 2 } }  set_temp_variable = { mcksp_tmp_ax = -1 }   set_temp_variable = { mcksp_tmp_ay = -4 } }
	else_if = { limit = { check_variable = { player_self = 3 } }  set_temp_variable = { mcksp_tmp_ax = -5 } set_temp_variable = { mcksp_tmp_ay = 1 } }
	else_if = { limit = { check_variable = { player_self = 4 } }  set_temp_variable = { mcksp_tmp_ax = 1 }     set_temp_variable = { mcksp_tmp_ay = 1 } }
	else_if = { limit = { check_variable = { player_self = 5 } }  set_temp_variable = { mcksp_tmp_ax = 2 }   set_temp_variable = { mcksp_tmp_ay = 2 } }
	else_if = { limit = { check_variable = { player_self = 6 } }  set_temp_variable = { mcksp_tmp_ax = 1 }     set_temp_variable = { mcksp_tmp_ay = 4 } }
	else_if = { limit = { check_variable = { player_self = 7 } }  set_temp_variable = { mcksp_tmp_ax = 2 }     set_temp_variable = { mcksp_tmp_ay = 6 } }
	else_if = { limit = { check_variable = { player_self = 8 } }  set_temp_variable = { mcksp_tmp_ax = 4 }   set_temp_variable = { mcksp_tmp_ay = 8 } }
	else_if = { limit = { check_variable = { player_self = 9 } }  set_temp_variable = { mcksp_tmp_ax = -4 }    set_temp_variable = { mcksp_tmp_ay = -2 } }
	else_if = { limit = { check_variable = { player_self = 10 } } set_temp_variable = { mcksp_tmp_ax = -3 }    set_temp_variable = { mcksp_tmp_ay = 5 } }
}

MCKSP_classify_arseny = {
	set_temp_variable = { mcksp_class_arseny = 0 }
	if = {
		limit = { check_variable = { coalition_of_arseny = 0 } NOT = { has_country_flag = mcksp_guest_gone_arseny } }
		set_temp_variable = { mcksp_tmp_bx = -1 }
		set_temp_variable = { mcksp_tmp_by = -4 }
		MCKSP_calc_dist2 = yes
		if = { limit = { check_variable = { mcksp_tmp_dist2 < 101 } } set_temp_variable = { mcksp_class_arseny = 1 } }
		if = { limit = { check_variable = { mcksp_tmp_dist2 < mcksp_close_dist2 } }     set_temp_variable = { mcksp_class_arseny = 2 } }
	}
}

# Метит флаги/золото игрока «мгновенными» при выборе фракции: 4 флага ядер + золото
# своего ядра появляются сразу (игрок кликнул на компасе — панель открыта), без выезда.
MCKSP_mark_self_instant_cores = {
	if = {
		limit = { has_country_flag = SOV_compass_open }
		set_country_flag = mcksp_self_instant_nikita
		set_country_flag = mcksp_self_instant_evgeny
		set_country_flag = mcksp_self_instant_stepan
		set_country_flag = mcksp_self_instant_demokrit
	}
}
# Если коалиция, в которую вступили (mcksp_tmp_joined_c), стала размера 2 — ореол её
# ЯДРА появляется впервые, метим ядро мгновенным (иначе ядро выехало бы в одиночку).
MCKSP_halo_instant_after_join = {
	if = { limit = { check_variable = { mcksp_tmp_joined_c = 1 } check_variable = { coalition_size_1 = 2 } } set_country_flag = mcksp_halo_instant_nikita }
	else_if = { limit = { check_variable = { mcksp_tmp_joined_c = 2 } check_variable = { coalition_size_2 = 2 } } set_country_flag = mcksp_halo_instant_evgeny }
	else_if = { limit = { check_variable = { mcksp_tmp_joined_c = 3 } check_variable = { coalition_size_3 = 2 } } set_country_flag = mcksp_halo_instant_stepan }
	else_if = { limit = { check_variable = { mcksp_tmp_joined_c = 4 } check_variable = { coalition_size_4 = 2 } } set_country_flag = mcksp_halo_instant_demokrit }
}
# По одному на вербуемого гостя. Метит мгновенным только если панель открыта (игрок

# Событийные оповещения о вступлении / формировании коалиции (struggle_event.2/.3).
# Зовётся из обоих путей join (MCKSP_recruit_assign_<g> ИИ + MCKSP_invite_<g> игрок),
# рядом с MCKSP_mark_instant_<g>. mcksp_join_guest = индекс гостя (как в компасе),
# mcksp_join_coalition = его коалиция. struggle_event.3 — если коалиция стала размера 2.
# ============================================================================
# ОЧЕРЕДЬ оповещений (по одному ивенту за раз, mcksp_join_in_progress). Иначе при
# синхронной вербовке несколько country_event'ов читают ОБЩИЕ mcksp_join_* уже затёртыми
# -> «съедаются». announce ставит в очередь (joinpending_<g> / formedpending_<k>) + drain.
# Приоритет в drain: сперва «коалиция сформирована» (.3), потом «гость вступил» (.2).
MCKSP_announce_join_arseny = { set_country_flag = mcksp_joinpending_arseny set_variable = { mcksp_joincoal_arseny = coalition_of_arseny } set_temp_variable = { mcksp_tmp_jc = coalition_of_arseny } MCKSP_mark_formed_pending = yes MCKSP_join_drain = yes }

MCKSP_announce_join_ivan   = { set_country_flag = mcksp_joinpending_ivan   set_variable = { mcksp_joincoal_ivan = coalition_of_ivan }     set_temp_variable = { mcksp_tmp_jc = coalition_of_ivan }   MCKSP_mark_formed_pending = yes MCKSP_join_drain = yes }
MCKSP_announce_join_ilya   = { set_country_flag = mcksp_joinpending_ilya   set_variable = { mcksp_joincoal_ilya = coalition_of_ilya }     set_temp_variable = { mcksp_tmp_jc = coalition_of_ilya }   MCKSP_mark_formed_pending = yes MCKSP_join_drain = yes }
MCKSP_announce_join_artem  = { set_country_flag = mcksp_joinpending_artem  set_variable = { mcksp_joincoal_artem = coalition_of_artem }   set_temp_variable = { mcksp_tmp_jc = coalition_of_artem }  MCKSP_mark_formed_pending = yes MCKSP_join_drain = yes }
MCKSP_announce_join_ilgam  = { set_country_flag = mcksp_joinpending_ilgam  set_variable = { mcksp_joincoal_ilgam = coalition_of_ilgam }   set_temp_variable = { mcksp_tmp_jc = coalition_of_ilgam }  MCKSP_mark_formed_pending = yes MCKSP_join_drain = yes }
# Если коалиция mcksp_tmp_jc только что стала размера 2 — отметить её «сформированной» в очередь.
MCKSP_mark_formed_pending = {
	if = { limit = { check_variable = { mcksp_tmp_jc = 1 } check_variable = { coalition_size_1 = 2 } } set_country_flag = mcksp_formedpending_1 }
	else_if = { limit = { check_variable = { mcksp_tmp_jc = 2 } check_variable = { coalition_size_2 = 2 } } set_country_flag = mcksp_formedpending_2 }
	else_if = { limit = { check_variable = { mcksp_tmp_jc = 3 } check_variable = { coalition_size_3 = 2 } } set_country_flag = mcksp_formedpending_3 }
	else_if = { limit = { check_variable = { mcksp_tmp_jc = 4 } check_variable = { coalition_size_4 = 2 } } set_country_flag = mcksp_formedpending_4 }
}
MCKSP_join_drain = {
	if = {
		limit = { NOT = { has_country_flag = mcksp_join_in_progress } }
		if = { limit = { has_country_flag = mcksp_formedpending_1 } set_variable = { mcksp_join_coalition = 1 } set_country_flag = mcksp_join_in_progress country_event = { id = struggle_event.3 hours = 2 } }
		else_if = { limit = { has_country_flag = mcksp_formedpending_2 } set_variable = { mcksp_join_coalition = 2 } set_country_flag = mcksp_join_in_progress country_event = { id = struggle_event.3 hours = 2 } }
		else_if = { limit = { has_country_flag = mcksp_formedpending_3 } set_variable = { mcksp_join_coalition = 3 } set_country_flag = mcksp_join_in_progress country_event = { id = struggle_event.3 hours = 2 } }
		else_if = { limit = { has_country_flag = mcksp_formedpending_4 } set_variable = { mcksp_join_coalition = 4 } set_country_flag = mcksp_join_in_progress country_event = { id = struggle_event.3 hours = 2 } }
		else_if = { limit = { has_country_flag = mcksp_joinpending_arseny } set_variable = { mcksp_join_guest = 2 }  set_variable = { mcksp_join_coalition = mcksp_joincoal_arseny } set_country_flag = mcksp_join_in_progress country_event = { id = struggle_event.2 hours = 2 } }
		else_if = { limit = { has_country_flag = mcksp_joinpending_gleb }   set_variable = { mcksp_join_guest = 4 }  set_variable = { mcksp_join_coalition = mcksp_joincoal_gleb }   set_country_flag = mcksp_join_in_progress country_event = { id = struggle_event.2 hours = 2 } }

# Поиск ближайшего достижимого свободного кандидата для ядра (ax/ay = коорд ядра).
# Результат: mcksp_tmp_best_guest (id 0=нет). Детерминированный — без хрупких
# random_list-весов по temp (прежний баг: коалиции не формировались).
MCKSP_find_best_candidate = {
	set_temp_variable = { mcksp_tmp_best_dist2 = 9999 }
	set_temp_variable = { mcksp_tmp_best_guest = 0 }
	if = { limit = { check_variable = { coalition_of_arseny = 0 } NOT = { has_country_flag = mcksp_guest_gone_arseny } NOT = { has_country_flag = mcksp_inviting_arseny } NOT = { has_country_flag = mcksp_being_recruited_arseny } }
		set_temp_variable = { mcksp_tmp_bx = -1 } set_temp_variable = { mcksp_tmp_by = -4 } MCKSP_calc_dist2 = yes
		if = { limit = { check_variable = { mcksp_tmp_dist2 < 101 } check_variable = { mcksp_tmp_dist2 < mcksp_tmp_best_dist2 } } set_temp_variable = { mcksp_tmp_best_dist2 = mcksp_tmp_dist2 } set_temp_variable = { mcksp_tmp_best_guest = 2 } } }
	if = { limit = { check_variable = { coalition_of_gleb = 0 } NOT = { has_country_flag = mcksp_guest_gone_gleb } NOT = { has_country_flag = mcksp_inviting_gleb } NOT = { has_country_flag = mcksp_being_recruited_gleb } }
		set_temp_variable = { mcksp_tmp_bx = 1 } set_temp_variable = { mcksp_tmp_by = 1 } MCKSP_calc_dist2 = yes
		if = { limit = { check_variable = { mcksp_tmp_dist2 < 101 } check_variable = { mcksp_tmp_dist2 < mcksp_tmp_best_dist2 } } set_temp_variable = { mcksp_tmp_best_dist2 = mcksp_tmp_dist2 } set_temp_variable = { mcksp_tmp_best_guest = 4 } } }
	if = { limit = { check_variable = { coalition_of_ivan = 0 } NOT = { has_country_flag = mcksp_guest_gone_ivan } NOT = { has_country_flag = mcksp_inviting_ivan } NOT = { has_country_flag = mcksp_being_recruited_ivan } }
		set_temp_variable = { mcksp_tmp_bx = 1 } set_temp_variable = { mcksp_tmp_by = 4 } MCKSP_calc_dist2 = yes
		if = { limit = { check_variable = { mcksp_tmp_dist2 < 101 } check_variable = { mcksp_tmp_dist2 < mcksp_tmp_best_dist2 } } set_temp_variable = { mcksp_tmp_best_dist2 = mcksp_tmp_dist2 } set_temp_variable = { mcksp_tmp_best_guest = 6 } } }
	if = { limit = { check_variable = { coalition_of_ilya = 0 } NOT = { has_country_flag = mcksp_guest_gone_ilya } NOT = { has_country_flag = mcksp_inviting_ilya } NOT = { has_country_flag = mcksp_being_recruited_ilya } }

		if = { limit = { check_variable = { mcksp_tmp_dist2 < 101 } check_variable = { mcksp_tmp_dist2 < mcksp_tmp_best_dist2 } } set_temp_variable = { mcksp_tmp_best_dist2 = mcksp_tmp_dist2 } set_temp_variable = { mcksp_tmp_best_guest = 7 } } }
	if = { limit = { check_variable = { coalition_of_artem = 0 } NOT = { has_country_flag = mcksp_guest_gone_artem } NOT = { has_country_flag = mcksp_inviting_artem } NOT = { has_country_flag = mcksp_being_recruited_artem } }
		set_temp_variable = { mcksp_tmp_bx = -4 } set_temp_variable = { mcksp_tmp_by = -2 } MCKSP_calc_dist2 = yes
		if = { limit = { check_variable = { mcksp_tmp_dist2 < 101 } check_variable = { mcksp_tmp_dist2 < mcksp_tmp_best_dist2 } } set_temp_variable = { mcksp_tmp_best_dist2 = mcksp_tmp_dist2 } set_temp_variable = { mcksp_tmp_best_guest = 9 } } }
