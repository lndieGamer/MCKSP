# Активируется в activate_guests_system. Edge ranges (±0.95..±1) триггерят
# trust_event.1 (ВКП(б)) и trust_event.2 (расстрел). Иконки сторон — vanilla ETH
# заглушки до своих gfx.

SOV_stalin_trust_balance = {
	initial_value = 0
	left_side = SOV_stalin_distrust_side
	right_side = SOV_stalin_trust_side

	# Решения механики показываются внутри окна баланса (powerbalanceview), а не
	# в обычной вкладке решений.
	decision_category = SOV_MCKSP_trust_cat

	range = {
		id = SOV_stalin_neutral_range
		min = -0.2
		max = 0.2
		modifier = {
			political_power_factor = 0
		}
	}

	side = {
		id = SOV_stalin_distrust_side
		icon = GFX_bop_ETH_decentralisation_side

		range = {
			id = SOV_stalin_distrust_mild_range
			min = -0.7
			max = -0.2
			modifier = {
				political_power_factor = -0.05
				stability_factor = -0.05
			}
		}

		range = {
			id = SOV_stalin_distrust_strong_range
			min = -0.95
			max = -0.7
			modifier = {
				political_power_factor = -0.1
				stability_factor = -0.1
				research_speed_factor = -0.05
			}
		}

		range = {
			id = SOV_stalin_distrust_terminal_range
			min = -1
			max = -0.95
			modifier = {
				political_power_factor = -0.15
				stability_factor = -0.15
				research_speed_factor = -0.1
			}
			on_activate = {
				hidden_effect = {
					IF = {
						limit = { NOT = { has_country_flag = SOV_guests_purged_flag } }
						country_event = {
							id = trust_event.2
							hours = 4
							random_hours = 8
						}
						set_country_flag = SOV_guests_purged_flag
					}
				}
			}
		}
	}

	side = {
		id = SOV_stalin_trust_side
		icon = GFX_bop_ETH_communist_side

		range = {
			id = SOV_stalin_trust_mild_range
			min = 0.2
			max = 0.7
			modifier = {
				political_power_factor = 0.05
				research_speed_factor = 0.02
			}
		}

		range = {
			id = SOV_stalin_trust_strong_range
			min = 0.7
			max = 0.95
			modifier = {
				political_power_factor = 0.1
				research_speed_factor = 0.05
				stability_factor = 0.05
			}
		}

		range = {
			id = SOV_stalin_trust_terminal_range
			min = 0.95
			max = 1
			modifier = {
				political_power_factor = 0.15
				research_speed_factor = 0.1
				stability_factor = 0.1
			}
			on_activate = {
				hidden_effect = {
					IF = {
						limit = { NOT = { has_country_flag = SOV_guests_joined_vkpb_flag } }
						country_event = {
							id = trust_event.1
							hours = 4
							random_hours = 8
						}
						set_country_flag = SOV_guests_joined_vkpb_flag
					}
				}
			}
		}
	}
}
