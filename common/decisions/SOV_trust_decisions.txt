				10 = { modifier = { factor = 0 OR = { NOT = { has_tech = infantry_weapons1 } has_tech = improved_infantry_weapons } } set_technology = { improved_infantry_weapons = 1 } }
				10 = { modifier = { factor = 0 OR = { NOT = { has_tech = improved_infantry_weapons } has_tech = advanced_infantry_weapons } } set_technology = { advanced_infantry_weapons = 1 } }
