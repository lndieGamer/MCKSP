# on_startup маскирует имена MCKSP-партий как MCKSP_redacted для ВСЕХ стран —
# scripted_localisation не работает в ideology-локалях, поэтому скрытие через
# party_name override. activate_guests_system снимает маскировку для SOV.
on_actions = {
	on_startup = {
		effect = {
			every_country = {
				set_party_name = { ideology = mcksp_social_liberalism long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_liberal_humanism long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_christian_democracy long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_market_socialism long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_pragmatic_syncretism long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_scientific_progressism long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_statist_technocratism long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_national_conservatism long_name = MCKSP_redacted name = MCKSP_redacted }
				set_party_name = { ideology = mcksp_democratic_socialism long_name = MCKSP_redacted name = MCKSP_redacted }
			}
		}
	}

	# Движок баланса доверия. Мёрджится с vanilla on_*_SOV (NSB), не перекрывает.
	# Тик еженедельный: шаг популярности 0.001 (0.1 п.п.) выше порога точности
	# popularity — дневной шаг 0.0001 округлялся бы вверх и дрейф был бы кратно
	# сильнее задуманного.
	on_weekly_SOV = {
		effect = {
			# Допрос Сталина ждёт завершения сокрытия инцидента Б: инцидент начат,
			# все something_*_happened сняты решениями сокрытия, допрос ещё не был.
			if = {
				limit = {
					has_country_flag = mcksp_incident_started
					NOT = { has_country_flag = mcksp_interrogation_done }
					NOT = { has_idea = something_maybe_happened_idea }
					NOT = { has_idea = something_happened_idea }
					NOT = { has_idea = something_definetely_happened_idea }
				}
				set_country_flag = mcksp_interrogation_done
				country_event = { id = interrogation_event.1 hours = 6 }
			}
			if = {
				# После вступления в ВКП(б) баланс доверия снят — пересчёт трона не нужен
				# (иначе add/remove_power_balance_modifier по снятому балансу = ошибки).
				limit = { has_country_flag = guests_active NOT = { has_country_flag = SOV_guests_in_vkpb } }
				MCKSP_recalc_cracked_throne = yes   # 1-в пересчёт тиров
			}
			if = {
				limit = {
					has_country_flag = guests_active
					has_idea = mcksp_secret_agitation_idea
				}
				MCKSP_agitation_tick = yes           # 2-г: +0.001/партию в неделю
			}

			# Фракционная борьба. Фаза 1 — авто-вербовка ядрами-ботами; фаза 2+ —
			# пересчёт популярностей коалиций для бар-чарта.
			if = {
				limit = { has_country_flag = guests_active check_variable = { mcksp_struggle_phase = 1 } }
				MCKSP_recruitment_tick = yes
				MCKSP_contest_drain = yes
				MCKSP_join_drain = yes
			}
			# Фаза 2: пуш ИИ гонят таймеры решений SOV_struggle_rival_push_* (как в фазе 1),
			# еженедельный тик не нужен.
			if = {
				limit = { has_country_flag = guests_active check_variable = { mcksp_struggle_phase > 0 } }
				MCKSP_publish_coalition_pops = yes
			}
		}
	}
}
